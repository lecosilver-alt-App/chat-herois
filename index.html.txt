            databaseURL: "https://chat-apk-alex-f-default-rtdb.firebaseio.com",
            projectId: "chat-apk-alex-f",

            messagingSenderId: "327918514106",
            appId: "1:327918514106:web:b61c63f4009b500bf67783"
        };
        
        // Inicializa o Firebase
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const chatBox = document.getElementById('chat-box');
        const msgInput = document.getElementById('msg-input');
        const nameInput = document.getElementById('name-input');

        // Função para enviar mensagem
        function sendMessage() {
            const text = msgInput.value.trim();
            const username = nameInput.value.trim() || "Anônimo";

            if (text !== "") {
                db.ref("mensagens").push({
                    nome: username,
                    texto: text,
                    timestamp: firebase.database.ServerValue.TIMESTAMP
                });
                msgInput.value = '';
            }
        }

        // Fica escutando novas mensagens
        db.ref("mensagens").on("child_added", function(snapshot) {
            const msgData = snapshot.val();
            
            const msgDiv = document.createElement('div');
            msgDiv.className = 'message';
            msgDiv.innerHTML = `<span class="username">${msgData.nome}</span> ${msgData.texto}`;
            
            chatBox.appendChild(msgDiv);
            chatBox.scrollTop = chatBox.scrollHeight;
        });
        
        // Enviar com a tecla "Enter"
        msgInput.addEventListener("keypress", function(event) {
            if (event.key === "Enter") {
                event.preventDefault();
                sendMessage();
            }
        });
